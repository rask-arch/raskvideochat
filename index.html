⚠️ Firestore: permessi mancanti

Le Security Rules di Firestore bloccano le scritture. Copia le regole qui sotto nel tuo progetto Firebase.

Vai su console.firebase.google.com → tuo progetto → Firestore Database → tab Rules → incolla questo e clicca Publish:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── Helper functions ──────────────────────────────────
    function isAuth()     { return request.auth != null; }
    function uid()        { return request.auth.uid; }
    function isOwner(id)  { return isAuth() && uid() == id; }
    function role()       {
      return isAuth()
        ? get(/databases/$(database)/documents/user_profiles/$(uid())).data.role
        : 'anon';
    }
    function isAdmin()    { return role() == 'Admin'; }
    function isStaff()    { return role() in ['Admin','Staff']; }
    function notBanned()  {
      return !('banned' in get(/databases/$(database)/documents/user_profiles/$(uid())).data)
          || get(/databases/$(database)/documents/user_profiles/$(uid())).data.banned == false;
    }

    // ── user_profiles ─────────────────────────────────────
    match /user_profiles/{userId} {
      allow read:   if isAuth();
      allow create: if isOwner(userId)
                    && request.resource.data.role in ['Guest','User']
                    && request.resource.data.name.size() >= 2
                    && request.resource.data.name.size() <= 30;
      allow update: if (isOwner(userId)
                        && request.resource.data.role == resource.data.role)
                    || isAdmin();
      allow delete: if isAdmin();
    }

    // ── online_users ──────────────────────────────────────
    match /online_users/{userId} {
      allow read:          if isAuth();
      allow create, update: if isOwner(userId) || isAdmin();
      allow delete:        if isOwner(userId) || isAdmin();
    }

    // ── messages ──────────────────────────────────────────
    match /messages/{msgId} {
      allow read:   if isAuth();
      allow create: if isAuth() && notBanned()
                    && request.resource.data.fromUid == uid()
                    && request.resource.data.text.size() <= 2000;
      allow update: if isStaff();
      allow delete: if isAdmin()
                    || (isStaff() && resource.data.fromUid != uid())
                    || resource.data.fromUid == uid();
    }

    // ── private_messages ──────────────────────────────────
    match /private_messages/{msgId} {
      allow read:   if isAuth()
                    && (resource.data.fromUid == uid()
                        || resource.data.toUid == uid());
      allow create: if isAuth() && notBanned()
                    && request.resource.data.fromUid == uid()
                    && request.resource.data.text.size() <= 2000;
      allow delete: if isAdmin()
                    || resource.data.fromUid == uid()
                    || resource.data.toUid == uid();
    }

    // ── rooms ─────────────────────────────────────────────
    match /rooms/{roomId}      { allow read: if isAuth(); allow write: if isAdmin(); }
    match /rooms_data/{docId}  { allow read: if isAuth(); allow write: if isStaff(); }

    // ── server_settings ───────────────────────────────────
    match /server_settings/{docId} {
      allow read: if true;   // config pubblica (colori, titolo)
      allow write: if isAdmin();
    }

    // ── cam / kicks / reactions ───────────────────────────
    match /cam_requests/{d}    { allow read, write: if isAuth(); }
    match /cam_viewers/{d}     { allow read, write: if isAuth(); }
    match /room_kicks/{d}      { allow read: if isAuth(); allow create,delete: if isStaff(); }
    match /live_reactions/{d}  { allow read, write: if isAuth(); }
    match /msg_reactions/{d}   { allow read: if isAuth(); allow write: if isAuth() && notBanned(); }
    match /typing_status/{d}   { allow read, write: if isAuth(); }

    // ── audit_log (immutabile) ────────────────────────────
    match /audit_log/{docId} {
      allow read:           if isAdmin();
      allow create:         if isStaff();
      allow update, delete: if false;
    }

    // ── Default deny ──────────────────────────────────────
    match /{document=**} { allow read, write: if false; }
  }
}

Puoi anche usare queste regole temporanee ⚠️ NON usare in produzione:
allow read, write: if true;
📌 Anche Firebase Authentication: Per gli ospiti abilita Anonymous in Authentication → Sign-in method.
